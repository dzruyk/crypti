
#array test script

arr = {5, 6, 1, 12, 4, 0, -2}
arr_len = 7


#insertion sort
for (i = 0; i < arr_len; i = i + 1) {
	min = i;
	for (j = i + 1; j < arr_len; j = j + 1) {
		if (arr[min] > arr[j]) {
			min = j
		}
	}
	if (min != i) {
		tmp = arr[min]
		arr[min] = arr[i]
		arr[i] = tmp
	}

}

arr

del(arr)

#buble sort
arr = {5, 6, 1, 12, 4, 0, -2}
arr_len = 7

while (1) {
	swapped = 0
	for (i = 1; i < arr_len; i = i + 1) {
			if (arr[i] < arr[i - 1]) {
				tmp = arr[i]
				arr[i] = arr[i - 1]
				arr[i - 1] = tmp
				swapped = 1;
			}
	}
	if (swapped == 0)
		break
}

arr

del(arr)
